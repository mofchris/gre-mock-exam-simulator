# Dark Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

## Goal

Add an auto + manual dark theme to both static study apps (GRE Mock Exam Simulator and Network+ Mock Exam Simulator) so night studying is not blinding. The theme follows the OS by default and a header toggle overrides it (cycle auto → dark → light). Everything themes; exam-taking surfaces render on an **elevated tier** (one step lighter than the page) to keep passages, DI charts, PBQs, and the calculator high-contrast. Light mode after this work must be a **pixel-level no-op**.

## Architecture

**Approach A** — JS-resolved theme attribute + a single dark token block per stylesheet:

1. A `js/theme.js` (byte-identical in both repos) loads in `<head>` **before** `css/style.css`, stamps an explicit `data-theme="dark"|"light"` on `<html>` pre-paint, and exposes a global `StudyTheme`.
2. Each stylesheet is **tokenized first** (mechanical, zero visual change): every color literal outside the token block is promoted to a CSS custom property. After this pass the sheet has zero color literals outside token blocks.
3. Each stylesheet then gets **one** `[data-theme="dark"] { … }` block that redefines those tokens with a real dark palette, including `color-scheme: dark`.
4. Each app's `chrome()` renders a glyph toggle button in `.topright`; a `≤720px` two-row header keeps the mobile chrome un-cramped.

sync.js and pbq.js are untouched (their UI is already tokenized; re-verified in Task 6). No new fonts, images, or non-color styling.

- Vanilla ES5-safe JS (IIFE, no build step, no dependencies), matching the existing `js/*.js` style

- GitHub Pages static hosting under `mofchris.github.io` (same-origin ⇒ `localStorage` theme choice is shared across both apps, like `study-sync-auth`)

- [ ] Create `C:/Users/mofch/OneDrive/Desktop/GRE/js/theme.js` with **exactly**:

```js
(function () {
  "use strict";

  var KEY = "study-theme";
  var mql = window.matchMedia ? window.matchMedia("(prefers-color-scheme: dark)") : null;
  var listeners = [];

  function read() {
    try {
      var v = localStorage.getItem(KEY);
      if (v === "dark" || v === "light") return v;
    } catch (e) {}
    return "auto";
  }

  function write(mode) {
    try {
      if (mode === "auto") localStorage.removeItem(KEY);
      else localStorage.setItem(KEY, mode);
    } catch (e) {}
  }

  function systemDark() {
    return !!(mql && mql.matches);
  }

  function resolve(mode) {
    if (mode === "dark") return "dark";
    if (mode === "light") return "light";
    return systemDark() ? "dark" : "light";
  }

  function apply(resolved) {
    document.documentElement.setAttribute("data-theme", resolved);
  }

  function current() {
    return resolve(read());
  }

  function emit() {
    var mode = read();
    var res = resolve(mode);
    for (var i = 0; i < listeners.length; i++) {
      try { listeners[i](mode, res); } catch (e) {}
    }
  }

  // Pre-paint: stamp the attribute the moment this script runs in <head>,
  // before the stylesheet is parsed, so there is no white flash at night.
  apply(current());

  // Auto mode live-follows the OS. In an overridden mode the media change is
  // ignored (we re-read every time and only re-apply while still "auto").
  function onMedia() {
    if (read() === "auto") { apply(current()); emit(); }
  }
  if (mql) {
    if (mql.addEventListener) mql.addEventListener("change", onMedia);
    else if (mql.addListener) mql.addListener(onMedia);
  }

  window.StudyTheme = {
    mode: function () { return read(); },
    resolved: function () { return current(); },
    cycle: function () {
      var order = { auto: "dark", dark: "light", light: "auto" };
      var next = order[read()] || "dark";
      write(next);
      apply(resolve(next));
      emit();
      return next;
    },
    onChange: function (fn) {
      if (typeof fn === "function") {
        listeners.push(fn);
        try { fn(read(), current()); } catch (e) {}
      }
    }
  };
})();
```

```js
(function () {
  "use strict";

  var KEY = "study-theme";
  var mql = window.matchMedia ? window.matchMedia("(prefers-color-scheme: dark)") : null;
  var listeners = [];

  function read() {
    try {
      var v = localStorage.getItem(KEY);
      if (v === "dark" || v === "light") return v;
    } catch (e) {}
    return "auto";
  }

  function write(mode) {
    try {
      if (mode === "auto") localStorage.removeItem(KEY);
      else localStorage.setItem(KEY, mode);
    } catch (e) {}
  }

  function systemDark() {
    return !!(mql && mql.matches);
  }

  function resolve(mode) {
    if (mode === "dark") return "dark";
    if (mode === "light") return "light";
    return systemDark() ? "dark" : "light";
  }

  function apply(resolved) {
    document.documentElement.setAttribute("data-theme", resolved);
  }

  function current() {
    return resolve(read());
  }

  function emit() {
    var mode = read();
    var res = resolve(mode);
    for (var i = 0; i < listeners.length; i++) {
      try { listeners[i](mode, res); } catch (e) {}
    }
  }

  // Pre-paint: stamp the attribute the moment this script runs in <head>,
  // before the stylesheet is parsed, so there is no white flash at night.
  apply(current());

  // Auto mode live-follows the OS. In an overridden mode the media change is
  // ignored (we re-read every time and only re-apply while still "auto").
  function onMedia() {
    if (read() === "auto") { apply(current()); emit(); }
  }
  if (mql) {
    if (mql.addEventListener) mql.addEventListener("change", onMedia);
    else if (mql.addListener) mql.addListener(onMedia);
  }

  window.StudyTheme = {
    mode: function () { return read(); },
    resolved: function () { return current(); },
    cycle: function () {
      var order = { auto: "dark", dark: "light", light: "auto" };
      var next = order[read()] || "dark";
      write(next);
      apply(resolve(next));
      emit();
      return next;
    },
    onChange: function (fn) {
      if (typeof fn === "function") {
        listeners.push(fn);
        try { fn(read(), current()); } catch (e) {}
      }
    }
  };
})();
```
